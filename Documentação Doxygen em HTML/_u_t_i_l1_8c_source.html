01034 **         dstSize         - Size of the destination buffer (in

01049   unsigned char i=0, j;
01050   unsigned char tmp;

01177 /*
01178 ** ===================================================================
01179 **     Method      :  UTIL1_ReadEscapedName (component Utility)
01180 **     Description :
01181 **         Scans an escaped name from a string. This is useful e.g. for
01182 **         double quoted file names.
01183 **     Parameters  :
01184 **         NAME            - DESCRIPTION
01185 **       * filename        - the name to be copied. Names may
01186 **                           be in quoted format
01187 **       * destname        - the destination of the copy.
01188 **                           Names are not in quoted format. destname
01189 **                           may be NULL to get the other return values
01190 **                           only
01191 **         maxlen          - length allocated for destname
01192 **       * lenRead         - length read in filename to copy
01193 **                           the whole name. Note that filenames maybe
01194 **                           space terminated, so *lenRead <
01195 **                           strlen(filename)
01196 **         lenWritten      - the size written in destname.
01197 **                           In case of overflows in destname,
01198 **                           lenWritten is still increased but destname
01199 **                           no longer written. The have the string
01200 **                           length in these cases use strlen(destname)
01201 **         terminators     - additional characters
01202 **                           where a name should terminate. May be NULL
01203 **     Returns     :
01204 **         ---             - Error code
01205 ** ===================================================================
01206 */
01207 uint8_t UTIL1_ReadEscapedName(const unsigned char *filename, uint8_t *destname, size_t maxlen, size_t *lenRead, size_t *lenWritten, const char *terminators)
01208 {
01209   size_t lenCopied = 0, lenOverread = 0;
01210   bool quoteMode = FALSE;  /* quoteMode means the name is surrounded by ". In this mode, only a second single quote "
01211                               terminates the string. In !quoteMode a space or a '\0' may also terminate it correctly */
01212   bool res = ERR_OK;
01213   #define IS_SPACE(ch) ((ch)==' '||(ch)=='\t'||(ch)=='\n'||(ch)=='\v'||(ch)=='\f'||(ch)=='\r')
01214 
01215   if (filename==NULL || (destname!=NULL && maxlen==0)) {
01216     return ERR_FAILED;
01217   }
01218   if (filename[0] == '"') { /* translated mode */
01219     filename++; /* overread '"' */
01220     lenOverread++;
01221     quoteMode=TRUE;
01222   }
01223   if (terminators == NULL) {
01224     terminators = "";
01225   }
01226   for (;;) {
01227     if (quoteMode) {
01228       if (filename[0] == '"') {
01229         filename++; /* overread '"' */
01230         lenOverread++;
01231         if (filename[0] != '"') { /* quoteMode is terminated by a single quote. A double quote is treated like a single quote and does not terminate it ! */
01232           break; /* successfully finished with this name */
01233         } /* else we copy the second quote " */
01234       }
01235       if (filename[0] == '\0') { /* unexpected 0. stop */
01236         res = ERR_FAILED;
01237         break; /* error case: no terminating double quote (") was found */
01238       }
01239     } else { /* copy mode */
01240       if (IS_SPACE(filename[0]) || filename[0] == '\0' || strchr(terminators, filename[0]) != NULL) { /* !quoteMode is terminated by space, '\0' or by any char in terminators */
01241         break;
01242       }
01243     }
01244     if (destname != NULL) {
01245       if (lenCopied + 1 < maxlen) {
01246         destname[0] = filename[0];
01247         destname++;
01248       } else {
01249         destname[0] = '\0'; /* terminate string */
01250         destname = NULL; /* avoid it to overwrite not allocated space */
01251       }
01252     }
01253     lenCopied++;
01254     filename++;
01255   }
01256   if (destname != NULL) {
01257     destname[0] = '\0';
01258   }
01259   if (lenRead != NULL) {
01260     *lenRead = lenCopied+lenOverread;
01261   }
01262   if (lenWritten != NULL) {
01263     *lenWritten = lenCopied + 1; /* additionally a zero byte written */
01264   }
01265   return res;
01266 }

01296 #endif
01297 uint8_t UTIL1_xatoi(const unsigned char **str, int32_t *res)
01298 {
01299 /* 123 -5   0x3ff 0b1111 0377 3.25  w "
01300       ^                               1st call returns 123 and next ptr
01301          ^                            2nd call returns -5 and next ptr
01302                  ^                    3rd call returns 1023 and next ptr
01303                         ^             4th call returns 15 and next ptr
01304                              ^        5th call returns 255 and next ptr
01305                                ^      6th call returns 3 and next ptr, caller needs to read '.'
01306                                  ^    7th call returns 25 and next ptr
01307                                     ^ 8th call fails and returns ERR_FAILED
01308 */
01309   uint32_t val;
01310   uint8_t c, r, s = 0;
01311 
01312   *res = 0;
01313   while (**str==' ') {
01314     (*str)++;                          /* Skip leading spaces */
01315   }
01316   c = **str;
01317   if (c == '-') {                      /* negative? */
01318     s = 1;
01319     c = *(++(*str));
01320   }
01321   if (c == '0') {
01322     c = *(++(*str));
01323     switch (c) {
01324       case 'x':                        /* hexadecimal */
01325         r = 16; c = *(++(*str));
01326         break;
01327       case 'b':                        /* binary */
01328         r = 2; c = *(++(*str));
01329         break;
01330       default:
01331         if (c <= ' ' || c == '.') {
01332           return ERR_OK;               /* single zero */
01333         }
01334         if (c < '0' || c > '9') {
01335           return ERR_FAILED;           /* invalid char */
01336         }
01337         r = 8;                         /* octal */
01338         break;
01339     } /* switch */
01340   } else {
01341     if (c < '0' || c > '9') {
01342       return ERR_FAILED;               /* EOL or invalid char */
01343     }
01344     r = 10;                            /* decimal */
01345   }
01346   val = 0;
01347   while (c > ' ' && c != '.') {
01348     if (c >= 'a') c -= 0x20;
01349     c -= '0';
01350     if (c >= 17) {
01351       c -= 7;
